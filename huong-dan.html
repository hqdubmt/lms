⚡ Hướng dẫn cài đặt nhanh

1

Cài đặt dependencies

cd /home/hqdu/quangdu/lms
npm install
2

Cấu hình environment

# API
cp apps/api/.env.example apps/api/.env

# Web
cp apps/web/.env.example apps/web/.env.local

# Deploy
cp deploy/backend/.env.example deploy/backend/.env
cp deploy/frontend/.env.example deploy/frontend/.env
3

Chạy services (Docker)

cd deploy
docker compose up -d postgres mongodb redis minio
4

Database migration & seed

cd apps/api
npx prisma migrate dev --name init
npx tsx prisma/seed.ts
5

Chạy môi trường development

# Terminal 1 — API (port 4000)
cd apps/api && npm run dev

# Terminal 2 — Web (port 3000)
cd apps/web && npm run dev
6

Deploy production

cd deploy
cp backend/.env.example backend/.env   # chỉnh giá trị thực
cp frontend/.env.example frontend/.env # chỉnh giá trị thực
docker compose up -d
